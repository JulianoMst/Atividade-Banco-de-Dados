

CREATE DATABASE Imobiliaria_sul;
USE Imobiliaria_sul;

CREATE TABLE Proprietario(
id_proprietario INT PRIMARY KEY,
nome VARCHAR(80),
telefone VARCHAR(20)
);

CREATE TABLE Imovel(
id_imovel INT PRIMARY KEY,
endereco VARCHAR (120),
tipo VARCHAR(30),
valor_mensal DECIMAL(10,2),
id_proprietario INT,
FOREIGN KEY (id_proprietario) REFERENCES Proprietario (id_proprietario)
);

CREATE TABLE Cliente(
id_cliente INT PRIMARY KEY,
nome VARCHAR (80),
telefone VARCHAR (20),
id_cliente_indicado INT,
FOREIGN KEY (id_cliente_indicado) REFERENCES Cliente (id_cliente)
);

CREATE TABLE CORRETOR(
id_corretor INT PRIMARY KEY,
nome VARCHAR(80),
creci VARCHAR(20) UNIQUE
);

CREATE TABLE Contrato_locacao(
id_contrato INT PRIMARY KEY,
id_imovel INT,
id_cliente INT,
id_corretor INT,
data_i DATE,
data_f DATE,
FOREIGN KEY (id_imovel) REFERENCES Imovel (id_imovel),
FOREIGN KEY (id_cliente) REFERENCES Cliente (id_cliente),
FOREIGN KEY (id_corretor) REFERENCES Corretor (id_corretor)
);

-- ANSI92

SELECT i.endereco, i.valor_mensal, p.nome
FROM imovel i
INNER JOIN proprietario p
ON i.id_proprietario = p.id_proprietario;

SELECT c.nome, c.telefone, c2.nome AS indicado_por
FROM cliente c
LEFT JOIN cliente c2
ON c.id_cliente_indicado = c2.id_cliente;

SELECT co.nome
FROM corretor co
LEFT JOIN contrato_locacao cl
ON co.id_corretor = cl.id_corretor;

-- AUTO-RELACIONAMENTO

SELECT c.nome AS Cliente,
       c.telefone,
       c2.nome AS IndicadoPor
FROM cliente c
LEFT JOIN cliente c2
ON c.id_cliente_indicado = c2.id_cliente;

-- RELACIONAMENTOS AVANCADOS
SELECT i.endereco, i.valor_mensal, c.nome, co.nome
FROM imovel i, contrato_locacao cl, cliente c, corretor co
WHERE i.id_imovel = cl.id_imovel
AND c.id_cliente = cl.id_cliente
AND co.id_corretor = cl.id_corretor;
